FRA_Pull_To_Centre = {
	if = {
		limit = {
			power_balance_value = {
				id = FRA_power_balance
				value < -0.1
			}
		}
		add_power_balance_modifier = {
			id = FRA_power_balance
			modifier = FRA_Centre_Pull1
		}
	}
	else_if = {
		limit = {
			power_balance_value = {
				id = FRA_power_balance
				value > 0.1
			}
		}
		add_power_balance_modifier = {
			id = FRA_power_balance
			modifier = FRA_Centre_Pull2
		}
	}
}

FRA_Remove_Pull = {
	remove_power_balance_modifier = {
		id = FRA_power_balance
		modifier = FRA_Centre_Pull1
	}
	remove_power_balance_modifier = {
		id = FRA_power_balance
		modifier = FRA_Centre_Pull2
	}
}
