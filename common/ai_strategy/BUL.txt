BUL_kill_serbia = {
    allowed = {
		original_tag = BUL
	}
	
	enable = {
		has_war_with = SER
	}
	
	abort = {
	    OR = {
		    surrender_progress > 0.10
			casualties > 100000
			SER = {
				has_capitulated = yes
			}
		}
	}
	
	ai_strategy = {
		type = front_unit_request
		tag = SER
		value = 10000
	}
	
	ai_strategy = {
		type = front_control
		tag = SER
		priority = 2000
		ordertype = front
		execution_type = rush
		execute_order = yes
	}
}
