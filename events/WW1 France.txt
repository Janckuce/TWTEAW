#Germany Asks for Peace
country_event = {
	id = ww1_france.13
	title = ww1_france.13.t
	desc = ww1_france_desc.13
	picture = GFX_report_event_german_troops_ww1

	is_triggered_only = yes

	option = {
		name = ww1_france.13.a
		ai_chance = {
			factor = 100
		}
		#white peace Germany
			every_country = {
				limit = {
					has_war_with = GER
				}
				white_peace = GER
				white_peace = BUL
				white_peace = AUS
				white_peace = POL
				white_peace = UKR
				white_peace = LIT
				white_peace = UBD
			}

			#Free Luxembourg
			if = {
				limit = {
					GER = { controls_state = 8 }
				}
				LUX = { transfer_state = 8 }
			}

			#Treaty of Versailles
			GER = {
				country_event = { id = ww1_germany.15 days = 70 }
			}

			news_event = ww1_news.47
		
			#Subjects Become Free leave_faction = yes
		    if = {
				limit = {
					POL = { is_in_faction_with = GER }
				}
				GER = { set_autonomy = { target = POL autonomy_state = autonomy_free } }
				
				POL = { leave_faction = yes }
			if = {
				limit = {
					UKR = { is_in_faction_with = GER }
				}
				GER = { set_autonomy = { target = UKR autonomy_state = autonomy_free } }
				
				UKR = { leave_faction = yes }
			}
			if = {
				limit = {
					LIT = { is_in_faction_with = GER }
				}
				GER = { set_autonomy = { target = LIT autonomy_state = autonomy_free } }
				
				LIT = { leave_faction = yes }
			}
			if = {
				limit = {
					UBD = { is_in_faction_with = GER }
				}
				GER = { set_autonomy = { target = UBD autonomy_state = autonomy_free } }
				
				UBD = { leave_faction = yes }
			}
		}
	}
	option = {
		name = ww1_france.13.b
		
		ai_chance = {
			base = 0
		}
	}
}
